DATABASE TABLES CREATED - COMPLETE LIST
================================================================================

TABLE 1: profiles (Normalized)
├─ id (uuid, PRIMARY KEY)
├─ user_id (uuid, UNIQUE FK to auth.users)
├─ display_name, social_link
├─ main_photo_url, photo2_url, photo3_url
├─ is_complete (boolean)
├─ 8 JSONB fields for onboarding answers
├─ Conversation/voice fields
├─ created_at, updated_at (with auto-trigger)
└─ Index: idx_profiles_user_id

TABLE 2: membership (New)
├─ id (uuid, PRIMARY KEY)
├─ user_id (uuid, UNIQUE FK)
├─ status (pending|approved|rejected|applied)
├─ referral_code_used, application_data (jsonb)
├─ applied_at, approved_at
├─ created_at, updated_at
└─ Indexes: user_id, status

TABLE 3: membership_applications (New)
├─ id (uuid, PRIMARY KEY)
├─ user_id (uuid, UNIQUE FK)
├─ form_data (jsonb)
├─ referral_code
├─ status (pending|approved|rejected)
├─ Review tracking fields
├─ created_at, updated_at
└─ Indexes: user_id, status

TABLE 4: members (New)
├─ id (uuid, PRIMARY KEY)
├─ user_id (uuid, UNIQUE FK)
├─ referral_code (UNIQUE)
├─ referrer_id (FK to auth.users)
├─ joined_at, active (boolean)
├─ created_at, updated_at
└─ Indexes: user_id, referral_code, active

TABLE 5: matches (New)
├─ id (uuid, PRIMARY KEY)
├─ user_id, matched_user_id (FK to auth.users)
├─ match_score, match_decision
├─ Compatibility data (strengths, tensions, dealbreakers as jsonb)
├─ match_reason, archetype_dynamic
├─ status (pending_review|introduced|accepted|declined|expired)
├─ created_at, updated_at
└─ Indexes: user_id, matched_user_id, status, created_at

TABLE 6: messages (New)
├─ id (uuid, PRIMARY KEY)
├─ match_id (FK to matches)
├─ sender_id (FK to auth.users)
├─ content (text)
├─ read_at (nullable)
├─ created_at
└─ Indexes: match_id, sender_id, created_at
